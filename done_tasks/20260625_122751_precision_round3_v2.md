### 2A. 辞書ロード関数追加（ファイル先頭付近）

```python
import json
from pathlib import Path

_ALIAS_CACHE: dict | None = None

def _load_skill_aliases() -> dict:
    global _ALIAS_CACHE
    if _ALIAS_CACHE is not None:
        return _ALIAS_CACHE
    json_path = Path(__file__).resolve().parent.parent / "matching_v3" / "skill_aliases.json"
    try:
        with open(json_path, "r", encoding="utf-8") as f:
            data = json.load(f)
        alias_map = {}
        # aliases: {"aws": "AWS", ...}
        for k, v in data.get("aliases", {}).items():
            alias_map[k.lower().strip()] = v
        # soft_aliases
        # soft_aliasesは常にロード（soft_aliases_enabled=trueが現在のデフォルト）
        for k, v in data.get("soft_aliases", {}).items():
            alias_map[k.lower().strip()] = v
        # canonical_skills自身も追加（自己参照）
        for c in data.get("canonical_skills", []):
            alias_map[c.lower().strip()] = c
        _ALIAS_CACHE = alias_map
        logger.info("Loaded %d skill aliases from %s", len(alias_map), json_path)
    except Exception as e:
        logger.warning("Failed to load skill_aliases.json: %s. Using empty aliases.", e)
        _ALIAS_CACHE = {}
    return _ALIAS_CACHE
```

### 3B. extract_skills()にLLM fallback追加

extract_skills()内、filter_skills()の後に:

```python
# LLM fallback: 尚可が空 + テキストにヒントワードあり + CostGuard許可
# body_norm は extract_skills() 内で既に定義済みの変数:
#   body_norm = unicodedata.normalize("NFKC", (body or "")[:2000])
OPTIONAL_HINT_WORDS = ['尚可', '歓迎', 'あれば', '優遇', 'プラス', 'nice to have', 'want']
if not valid_optional and any(w in body_norm.lower() for w in OPTIONAL_HINT_WORDS):
    # CostGuardはモジュール先頭でimport（テストでパッチ可能にする）:
    #   from matching_cost_guard import CostGuard  ← ファイル先頭に追加
    cg = CostGuard()
    if cg.can_call(input_tokens=len(body_norm)//4, output_tokens=200, target_id="optional_extract"):
        try:
            valid_optional = _llm_extract_optional(body_norm, aliases)
        except Exception as e:
            logger.warning("LLM optional extraction failed: %s", e)
```

_llm_extract_optional(body_text: str, aliases: dict) の実装:

```python
from openai import OpenAI
import json as json_mod

def _llm_extract_optional(body_text: str, aliases: dict) -> list:
    # 既存パターン: matching_v3/matching_v3.py L45-322 参照
    client = OpenAI()  # config/.env の OPENAI_API_KEY を使用
    resp = client.chat.completions.create(
        model="gpt-4.1-nano",
        messages=[
            {"role": "system", "content": "SESメールから尚可スキルのみをJSON配列で返せ。必須スキルは含めるな。原子的なスキル名のみ。該当なしは[]。"},
            {"role": "user", "content": body_text[:1500]}
        ],
        max_tokens=300,
        temperature=0,
    )
    text = resp.choices[0].message.content.strip()
    text = text.replace("```json", "").replace("```", "").strip()
    skills = json_mod.loads(text)
    if not isinstance(skills, list):
        return []
    return filter_skills(skills, aliases=aliases)
```

CostGuard記録: cg.record_call()（can_call成功後に自動記録される既存パターンに従う）
